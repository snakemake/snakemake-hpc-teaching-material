\begin{frame}[fragile]
  \frametitle{Which Editor or IDE should I be using?}
  \begin{hint}
     You may use any \emph{any} editor of your liking..
  \end{hint}
  
  On \texttt{Marvin} you have the following options:
  \begin{itemize}[<+->]
	\item \altverb{nano} - a terminal-based beginner editor
	\item \altverb{vi} resp. \altverb{vim} - a terminal-based pro editor
	\item if you brought your own device, you might use \altverb{VSC} per sshfs
  \end{itemize}
\end{frame}
